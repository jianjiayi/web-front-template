### 防抖
```bash
import { Debounce } from 'lodash-decorators';
/**
   * Debounce 防抖
   * 防止多次点击频繁触发
   * @ decorators 装饰器
   * 只适用与类和类的属性
   * 不适用于箭头函数
   * 使用方法：this.handleSubmit.bind(this)
   * @param {*} err
   * @param {*} values
   * @memberof LoginPage
   */
  @Debounce(200)
  handleSubmit(err, values) {
    // const { type } = this.state;
    if (!err) {
      this.login(values)
    }
  };
```
